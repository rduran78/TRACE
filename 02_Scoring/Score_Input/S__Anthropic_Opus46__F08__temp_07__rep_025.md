    if (length(idx) == 0L) next

    out[j, 2L] <- min(nv)
    out[j, 3L] <- mean(nv)
  }
  out
}

  min_col  <- paste0("neighbor_min_",  var_name)

  for (var_name in neighbor_source_vars) {
